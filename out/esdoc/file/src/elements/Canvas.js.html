    src/elements/Canvas.js

    import Element from './Element'

class Canvas extends Element {
    constructor() {
        super('canvas')
        this._context = this.getContext2D()
    }
    getContext() {
        return this._context
    }
    setContext(ctx) {
        return this._context = this.getContext(ctx)
    }
    getContext2D() {
        return this.getElement().getContext('2d')
    }
    getContext3D() {
        return this.getElement().getContext('3d')
    }
    drawLine(beginX, beginY, endX, endY) {
        let ctx = this.getContext()
        ctx.moveTo(beginX, beginY)
        ctx.lineTo(endX, endY)
        ctx.stroke()
    }
    drawSquare(x, y, width, height, type) {
        let ctx = this.getContext()
        switch (type) {
            case 'stroke':
                ctx.strokeRect(x, y, width, height, type)
                break
            case 'clear':
                ctx.clearRect(x, y, width, height, type)
                break
            default:
                ctx.fillRect(x, y, width, height, type)
        }
    }
    drawTriangle(x, y, x1, y1, x2, y2, type) {
        let ctx = this.getContext()
        ctx.beginPath();
        ctx.moveTo(x, y);
        ctx.lineTo(x1, y1);
        ctx.lineTo(x2, y2);

        switch (type) {
            case 'stroke':
                ctx.stroke()
                break
            case 'clear':
                ctx.clear()
                break
            default:
                ctx.fill();
        }
    }
    drawTriangleWithDimension(x, y, dimension, type) {
        let ctx = this.getContext()
        ctx.beginPath();
        ctx.moveTo(x, y);
        ctx.lineTo(x1-dimension, y1+dimension);
        ctx.lineTo(x2+dimension, y2+dimension);

        switch (type) {
            case 'stroke':
                ctx.stroke()
                break
            case 'clear':
                ctx.clear()
                break
            default:
                ctx.fill();
        }
    }
    beginPath() {
        let ctx = this.getContext()
        return ctx.beginPath()
    }
    closePath() {
        let ctx = this.getContext()
        return ctx.closePath()
    }
    stroke() {
        let ctx = this.getContext()
        return ctx.stroke()
    }
    fill() {
        let ctx = this.getContext()
        return ctx.fill()
    }

};

export default Canvas
